#include <bits/stdc++.h>
using namespace std;
const int maxn = 1000 + 5;

int n, m;
int h[maxn][maxn];
int q[maxn], l[maxn], r[maxn];

int sol(int l[], int a[]){
    a[0] = -1;
    int t = 0;
    for(int i = 1; i <= m; i ++ ) {
        while(a[q[t]] >= a[i]) t --;
        l[i] = q[t] + 1;
        q[ ++ t ] = i;
    }
}

int work(int a[]){
    sol(l, a);
    reverse(a + 1, a + 1 + m);
    sol(r, a);
    reverse(a + 1, a + 1 + m);
    int res = 0;
    for(int i = 1; i <= m; i ++ ) {
        int le = l[i];
        int ri = m + 1 - r[m - i + 1];
        res = max(res, a[i] * (ri - le + 1));
    }
    return res;
}

char mp[maxn][maxn];

int main(){
    cin >> n >> m;
    int ans = 0;
    for(int i = 1; i <= n; i ++ ) {
        for(int j = 1; j <=m; j ++ ) {

            if(mp[i][j] == 'F') h[i][j] = h[i - 1][j] + 1;
            else continue;
        }
        ans = max(ans, work(h[i]));
    }
    
    cout << ans * 3 << endl;
    return 0;
}

#include <iostream>

using namespace std;

string str_mins(string s){
    int n = s.size();
    for(int i = 0; i < n; i ++) s.push_back(s[i]);
    int i = 0, j = 1, k;
    while( i < n && j < n){
        for(k = 0; k < n && s[i + k] == s[j + k]; k ++ );
        if(k == n) break;
        if(s[i + k] > s[j + k]) {
            i = i + k + 1;
            if(i == j) i ++;
        }else{
            j = j + k + 1;
            if(i == j) j ++;
        }
    }
    string res;
    k = min(i, j);
    for(int i = k; i < n + k; i ++) res += s[i];
    return res;
}

    s1 = str_mins(s1);

bool chk(int mid, int i) {
    ULL ss2 = h2[mid];
    ULL ss1 = h1[i + mid - 1] - h1[i - 1] * p[mid];
    if(ss2 != ss1) return 0;
    else return 1;
}

int main() {
    int n, m, q;

    cin >> (s1 + 1) >> (s2 + 1) ;

        h1[i] = h1[i - 1] * 131 + s1[i];
        p[i] = p[i - 1] * 131;
    }
    
    for(int i = 1 ; i <= m; i ++) {
        h2[i] = h2[i - 1] * 131 + s2[i];
        p[i] = p[i - 1] * 131;
    }
    
    for(int i = 1; i <= n; i ++) {
        int l = 0, r = min(n, m);
        while(l < r){
            int mid = l + r + 1 >> 1;

int trie[maxn << 1][12], tot = 1;

bool cmp(const string &a, const string &b){
    return a.size() < b.size() || ( a.size() == b.size() && a < b );
}

void ins(string s){
    int p = 1, len = s.size();
    for(int k = 0; k < len; k ++ ) {
        int ch = s[k] - '0';
        if(trie[p][ch] == 0) trie[p][ch] = ++ tot;
        p = trie[p][ch];

}

int ask(string s){
    int p = 1, len = s.size();

}

#include <bits/stdc++.h>

priority_queue<int, vector<int>, greater<int> > minheap;
priority_queue<int, vector<int>, less<int> > maxheap;

const int maxn = 3e4 + 5;

int a[maxn];
int vis[maxn];
int k, n, m;

int main(){
    cin >> n >> m;   
    for(int i = 1; i <= n; i ++) {
        cin >> a[i];
    }    
    for(int i = 1, a; i <= m ; i++ ) {
        cin >> a;
        vis[a]++;
    }

        maxheap.push(a[i]);
        if(maxheap.size() > k) {
