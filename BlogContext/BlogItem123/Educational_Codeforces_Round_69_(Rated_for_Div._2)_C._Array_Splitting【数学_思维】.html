#include <bits/stdc++.h>
using namespace std;

int cas, n, m , k;
int a[300005];
int b[300005];
signed main() {
	int i, j;
	cin >> n >> k;
	for (i = 0; i < n; i++)
		cin >> a[i];
	for (i = 0; i < n; i++)
		b[i] = a[i] - a[i - 1];

	sort(b, b + n - 1, greater<int>{});
	int ans = a[n - 1] - a[0];
	for (i = 0; i < k - 1; i++)
		ans -= b[i];

	cout << ans << endl;

    return 0;
}
