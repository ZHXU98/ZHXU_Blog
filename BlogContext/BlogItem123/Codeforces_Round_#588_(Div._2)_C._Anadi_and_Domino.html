#include <bits/stdc++.h>
using namespace std;
const int maxn = 50;

int n, m, ans;
int x[maxn], y[maxn];
int val[maxn];
bool vis[maxn][maxn];

void dfs(int pos) {
    if(pos > n) {
        int tmp = 0;
        memset(vis, 0, sizeof vis);
        for(int i = 1; i <= m; i++) {
            int a = val[x[i]], b = val[y[i]];
            if(!vis[a][b]) {
                tmp ++;
                vis[a][b] = vis[b][a] = 1;
            }
        }
        ans = max(tmp, ans);
        return ;
    }
    for(int i = 1; i <= 6; i++) {
        val[pos] = i;
        dfs(pos + 1);
    }
}

int main() {
    cin >> n >> m;
    // 题解也是nb啊  n <= 6的时候就是m个  6个完全够放不同的  最多15边
    for(int i = 1, a, b; i <= m ;i ++)
        cin >> x[i] >> y[i];
    dfs(1);
    cout << ans <<endl;
	return 0;
}
